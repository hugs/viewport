//Width and height
var unit_of_measure = "mm"
var w = "200" + unit_of_measure
var h = "50" + unit_of_measure


var svg = d3.select("#viewport")
            .append("svg")
            .attr("id", "svg")
            .attr("width", w)
            .attr("height", h);

var dataset = [ 8, 16, 24, 32, 40, 48, 56];
 
var circles = svg.selectAll("circle")
                 .data(dataset)
                 .enter()
                 .append("circle");

circles.attr("cx", function(d, i) {
            return d + unit_of_measure;
        })
       .attr("cy", "25" + unit_of_measure)
       .attr("r", "2.4" + unit_of_measure)


var node = document.getElementById('svg')
var serializer = new XMLSerializer()
var serialized = serializer.serializeToString(node)

var pretty = vkbeautify.xml(serialized,4); 
var text = d3.select("#xmloutput")
             .text(pretty);
